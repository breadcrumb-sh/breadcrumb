## Setup

Initialize the Breadcrumb client with your API key. You can get one from your [project settings](/docs/product/project-settings) after [setting up](/docs/setup/hosted) your instance.

<CodeBlockTabs defaultValue="ai-sdk">
  <CodeBlockTabsList>
    <CodeBlockTabsTrigger value="ai-sdk">AI SDK</CodeBlockTabsTrigger>
    <CodeBlockTabsTrigger value="manual">Manual</CodeBlockTabsTrigger>
  </CodeBlockTabsList>
  <CodeBlockTab value="ai-sdk">

```ts
import { init } from "@breadcrumb-sdk/core";
import { initAiSdk } from "@breadcrumb-sdk/ai-sdk";

const bc = init({
  apiKey: process.env.BREADCRUMB_API_KEY!,
  baseUrl: process.env.BREADCRUMB_BASE_URL!,
});

const { telemetry } = initAiSdk(bc);
```

## Start Tracing

<CodeBlockTabs defaultValue="ai-sdk">
  <CodeBlockTabsList>
    <CodeBlockTabsTrigger value="ai-sdk">AI SDK</CodeBlockTabsTrigger>
    <CodeBlockTabsTrigger value="manual">Manual</CodeBlockTabsTrigger>
  </CodeBlockTabsList>
  <CodeBlockTab value="ai-sdk">

```ts
const { text } = await generateText({
  model: anthropic("claude-sonnet-4-5"),
  prompt: "Explain what Breadcrumb does in one sentence.",

  // enable traces for this call
  experimental_telemetry: telemetry("quickstart"),
});
```

  </CodeBlockTab>
  <CodeBlockTab value="manual">

```ts
await bc.trace("quickstart", async (root) => {
  root.set({ input: prompt });

  const result = await bc.span(
    "generate",
    async (span) => {
      const text = await callYourModel(prompt);

      span.set({
        input: prompt,
        output: text,
        model: "your-model",
        provider: "your-provider",
      });

      return text;
    },
    { type: "llm" }
  );

  root.set({ output: result });
});
```

  </CodeBlockTab>
</CodeBlockTabs>

Open your Breadcrumb dashboard to see the trace.
